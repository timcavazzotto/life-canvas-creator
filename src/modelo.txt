/* ═══ DEMO GRID ═══ */
function buildDemoGrid() {
  const g = document.getElementById('demoGrid');
  g.innerHTML = '';
  const lived = Math.floor((Date.now() - new Date(1985,0,1)) / 6048e5);
  const total = 80 * WEEKS;
  for (let i = 0; i < total; i++) {
    const c = document.createElement('div');
    c.className = 'dp-cell ' + (i < lived ? 'lived' : 'future');
    g.appendChild(c);
  }
}

/* ═══ MAIN GRID ═══ */
function buildGrid() {
  const total = st.expect * WEEKS;
  const lived = st.birth ? Math.min(Math.floor((Date.now()-new Date(st.birth,0,1))/6048e5), total) : 0;
  const ms = MONTHS[st.lang];

  // decade col
  const dc = document.getElementById('decadeCol');
  dc.innerHTML = '';
  const rh = 10.3;
  for (let d = 0; d < Math.ceil(st.expect/10); d++) {
    const e = document.createElement('div');
    e.className = 'dec-lbl';
    const rows = Math.min(10, st.expect - d*10);
    e.style.height = `${rows * rh + (d > 0 ? 4 : 0)}px`;
    e.style.paddingTop = d > 0 ? '4px' : '0';
    e.textContent = d * 10;
    dc.appendChild(e);
  }

  // month row
  const mr = document.getElementById('monthRow');
  mr.innerHTML = '';
  for (let w = 0; w < WEEKS; w++) {
    const e = document.createElement('div');
    e.className = 'm-lbl';
    const mi = WEEK_POS.indexOf(w);
    if (mi >= 0) e.textContent = ms[mi];
    mr.appendChild(e);
  }

  // year rows
  const yr = document.getElementById('yearRows');
  yr.innerHTML = '';
  for (let y = 0; y < st.expect; y++) {
    const row = document.createElement('div');
    row.className = 'yr' + (y > 0 && y % 10 === 0 ? ' dec-sep' : '');
    for (let w = 0; w < WEEKS; w++) {
      const c = document.createElement('div');
      c.className = 'wk ' + (y*WEEKS+w < lived ? 'lived' : 'future');
      row.appendChild(c);
    }
    yr.appendChild(row);
  }

  // stats
  const left = Math.max(0, total - lived);
  const pct = lived > 0 ? Math.round(lived/total*100) : 0;
  document.getElementById('sLived').textContent = lived > 0 ? lived.toLocaleString('pt-BR') : '—';
  document.getElementById('sLeft').textContent  = lived > 0 ? left.toLocaleString('pt-BR') : '—';
  document.getElementById('sTotal').textContent = total.toLocaleString('pt-BR');
  document.getElementById('sPct').textContent   = lived > 0 ? pct+'%' : '—';
  document.getElementById('cfgBar').style.width = pct+'%';
  document.getElementById('pfTotal').textContent = total.toLocaleString('pt-BR');
  document.getElementById('pfLived').textContent = lived > 0 ? lived.toLocaleString('pt-BR') : '—';
  document.getElementById('pfLeft').textContent  = lived > 0 ? left.toLocaleString('pt-BR') : '—';
}

  document.getElementById('pEyebrow').textContent    = t.eyebrow[l];
  document.getElementById('pQuote').textContent      = t.quote[l];
  document.getElementById('pAttr').textContent       = t.attr[l];
  document.getElementById('pLegNote').innerHTML      = t.note[l];
  document.getElementById('pTagline').textContent    = t.tag[l];
  document.getElementById('pSubtitle').textContent   = `${(st.expect*WEEKS).toLocaleString('pt-BR')} ${wk} · ${st.expect} ${yr} · ${al}`;
  document.getElementById('pName').textContent       = st.name || '\u00a0';
  document.getElementById('pBirth').textContent      = st.birth || '\u00a0';

  document.getElementById('lbNome').textContent      = lb.nome;

  document.getElementById('lbExp').textContent       = lb.exp;
  document.getElementById('lbDedic').textContent     = lb.dedic;
  document.getElementById('lbLvd').textContent       = lb.lvd;

}
